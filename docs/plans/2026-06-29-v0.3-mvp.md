# WeHireMonitor v0.3 MVP 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 接入 VLM(Qwen-VL)处理长图招聘海报,实现长图切片+VLM提取+切片合并去重+每日预算硬上限+人工复核队列 CLI,将 OCR 低质文章从"标记 need_review 后丢弃"升级为"VLM 兜底提取"。

**Architecture:** 在 v0.2 双路切换基础上扩展为三路:`文本LLM(高质量) → OCR文本+LLM(中质量) → VLM切片提取(低质量/长图)`。新增 `providers/vlm/` 抽象层(Qwen-VL 实现+VLM Prompt)、`extractor/stitcher.py`(短图拼接为长图)、`extractor/slicer.py`(长图切片)、`extractor/budget.py`(预算管理器)、`extractor/vlm_merge.py`(切片合并去重)。Pipeline Runner 增加 VLM 调用计数和成本估算。CLI 新增 `review` 命令和 `--stats` 选项。

**现实场景处理:**
- 发布者将长图裁剪为多张短图发布 → `stitcher.py` 自动检测并拼接
- 拼接边界文字截断 → OCR 重叠区域检测+去重
- 重复图片(微信偶尔重复) → 感知哈希去重
- SVG 图片无法处理 → 跳过并记录警告
- 不同宽度图片不拼接 → 宽度差异检测

**Tech Stack:** 新增依赖: `Pillow>=10.0`(图片切片,复用已有)、`imagehash>=4.0`(图片感知哈希去重)。VLM 默认 Qwen-VL(`qwen-vl-max`)通过 DashScope API(OpenAI 兼容)。Prompt 模板外置为 `providers/prompts/vlm.txt`。

**依赖:** v0.2 已完成并验证通过(122 测试通过,DeepSeek LLM + RapidOCR 端到端验证)。

```python
"""Qwen-VL Provider 实现(DashScope API, OpenAI 兼容)

通过 DashScope 的 OpenAI 兼容接口调用 Qwen-VL-Max 模型。
API 文档: https://help.aliyun.com/zh/dashscope/developer-reference/use-qwen-by-calling-api

VLM Prompt 模板外置为 providers/prompts/vlm.txt(PRD §8.3.4)。
"""
from __future__ import annotations

import json
from pathlib import Path

import httpx
from loguru import logger

from wehire_monitor.domain.models import (
    ExtractionResult, Job, Deadline, ImageSlice,
)
from wehire_monitor.providers.vlm.base import VLMProvider, VLMResponse

_PROJECT_ROOT = Path(__file__).resolve().parent.parent.parent.parent
_PROMPT_PATH = _PROJECT_ROOT / "src" / "wehire_monitor" / "providers" / "prompts" / "vlm.txt"

# Qwen-VL-Max 定价: 约 0.02 元/千 token (输入) + 0.02 元/千 token (输出)
# 单次切片估算: ~1000 输入 token + ~500 输出 token ≈ 0.03 元
_COST_PER_SLICE = 0.03

# 重试参数
_MAX_ATTEMPTS = 3


class QwenVLProvider:
    """Qwen-VL 视觉语言模型实现"""

    name = "qwen_vl"

    def __init__(
        self,
        api_key: str,
        model: str = "qwen-vl-max",
        base_url: str = "https://dashscope.aliyuncs.com/compatible-mode/v1",
    ):
        self.api_key = api_key
        self.model = model
        self.base_url = base_url
        self.client = httpx.Client(timeout=60.0)
        self._prompt_template = self._load_prompt()

    def _load_prompt(self) -> str:
        if _PROMPT_PATH.exists():
            return _PROMPT_PATH.read_text(encoding="utf-8")
        # 兜底 prompt
        return (
            "你正在识别微信公众号招聘长图。只抽取图片中明确可见的内容。\n"
            "返回严格 JSON。\n"
            "article_title={{title}}\nimage_index={{image_index}}\n"
            "slice_index={{slice_index}}\ny_range={{y_start}}-{{y_end}}"
        )

    def _render_prompt(
        self, title: str, image_index: int, slice_index: int,
        y_start: int, y_end: int,
    ) -> str:
        prompt = self._prompt_template
        prompt = prompt.replace("{{title}}", title or "")
        prompt = prompt.replace("{{image_index}}", str(image_index))
        prompt = prompt.replace("{{slice_index}}", str(slice_index))
        prompt = prompt.replace("{{y_start}}", str(y_start))
        prompt = prompt.replace("{{y_end}}", str(y_end))
        return prompt

    def extract_jobs_from_slices(
        self,
        slices: list[ImageSlice],
        title: str,
        publish_time: str,
    ) -> VLMResponse:
        """从图片切片中提取岗位信息

        对每个切片调用 VLM,合并结果。成本 = 切片数 × _COST_PER_SLICE。
        """
        all_jobs: list[Job] = []
        all_warnings: list[str] = []
        total_cost = 0.0

        for sl in slices:
            meta = sl.meta
            prompt = self._render_prompt(
                title=title,
                image_index=meta.image_index,
                slice_index=meta.slice_index,
                y_start=meta.y_start,
                y_end=meta.y_end,
            )

            # 构造 OpenAI 兼容请求(带图片)
            image_url = self._get_image_url(sl)
            payload = {
                "model": self.model,
                "messages": [
                    {"role": "system", "content": "你是招聘信息结构化抽取助手。只从图片中抽取明确可见的信息。返回严格 JSON。"},
                    {"role": "user", "content": [
                        {"type": "text", "text": prompt},
                        {"type": "image_url", "image_url": {"url": image_url}},
                    ]},
                ],
                "temperature": 0.1,
            }

            response_data = self._call_api(payload)
            if response_data is None:
                all_warnings.append(f"slice_{meta.slice_index}: VLM API 失败")
                continue

            parsed = self._parse_response(response_data)
            if parsed is None:
                all_warnings.append(f"slice_{meta.slice_index}: JSON 解析失败")
                continue

            all_jobs.extend(parsed.get("jobs", []))
            all_warnings.extend(parsed.get("warnings", []))
            total_cost += _COST_PER_SLICE

        return VLMResponse(
            success=len(all_jobs) > 0 or not all_warnings,
            article_type="social_recruitment" if all_jobs else "unknown",
            jobs=all_jobs,
            warnings=all_warnings,
            cost_estimate=total_cost,
        )

    def _get_image_url(self, sl: ImageSlice) -> str:
        """获取图片 URL(base64 编码本地文件)"""
        import base64
        path = Path(sl.local_path)
        if path.exists():
            data = path.read_bytes()
            b64 = base64.b64encode(data).decode()
            return f"data:image/jpeg;base64,{b64}"
        return sl.local_path  # 兜底:可能是远程 URL

    def _call_api(self, payload: dict) -> dict | None:
        """调用 VLM API,带重试"""
        for attempt in range(_MAX_ATTEMPTS):
            try:
                resp = self.client.post(
                    f"{self.base_url}/chat/completions",
                    json=payload,
                    headers={
                        "Authorization": f"Bearer {self.api_key}",
                        "Content-Type": "application/json",
                    },
                )
                resp.raise_for_status()
                return resp.json()
            except httpx.HTTPStatusError as e:
                logger.warning(f"VLM API HTTP 错误 (attempt {attempt+1}): {e}")
                if attempt < _MAX_ATTEMPTS - 1:
                    import time
                    time.sleep(2 ** attempt)
            except (httpx.TimeoutException, httpx.HTTPError) as e:
                logger.warning(f"VLM API 网络错误 (attempt {attempt+1}): {e}")
                if attempt < _MAX_ATTEMPTS - 1:
                    import time
                    time.sleep(2 ** attempt)
            except Exception as e:
                logger.error(f"VLM API 未知错误: {e}")
                break
        return None

    def _parse_response(self, data: dict) -> dict | None:
        """解析 VLM 返回的 JSON"""
        try:
            content = data["choices"][0]["message"]["content"]
            # Qwen-VL 可能返回带 markdown 的 JSON,提取纯 JSON
            if "```json" in content:
                content = content.split("```json")[1].split("```")[0].strip()
            elif "```" in content:
                content = content.split("```")[1].split("```")[0].strip()
            return json.loads(content)
        except (KeyError, IndexError, json.JSONDecodeError) as e:
            logger.error(f"VLM JSON 解析失败: {e}")
            return None

    def close(self) -> None:
        self.client.close()
```

```python
"""长图识别与切片(PRD §8.3.2 / SRS §4.4)

切片策略:
- 单图高度 >3000px 且 OCR 文本 <300字 → 触发切片
- 最大切片高度 1800px(区间 1600-2200)
- 重叠 220px(区间 180-250)
- 原宽 >1440px 等比缩放到 1440px
- 每切片元信息: image_index/slice_index/y_start/y_end
- 底部含邮箱/二维码/报名方式切片强制保留(标记 is_bottom)
- 每篇最多 8 切片,超过先 OCR 找关键区域
"""
from __future__ import annotations

from pathlib import Path

from PIL import Image
from loguru import logger

from wehire_monitor.domain.models import ImageSlice, SliceMeta

# 切片参数(PRD §8.3.2)
SLICE_HEIGHT = 1800       # 最大切片高度
SLICE_OVERLAP = 220       # 重叠区域
MAX_WIDTH = 1440           # 最大宽度(超过等比缩放)
MAX_SLICES = 8             # 每篇最多切片数
LONG_IMAGE_THRESHOLD = 3000  # 长图高度阈值
LOW_OCR_TEXT_THRESHOLD = 300  # OCR 文本过少阈值


def should_slice_image(image_path: str | Path, ocr_text_len: int) -> bool:
    """判断是否需要切片

    触发条件: 单图高度 >3000px 且 OCR 文本 <300字
    (PRD §8.3.1 VLM 触发条件)
    """
    try:
        with Image.open(image_path) as img:
            height = img.height
    except Exception as e:
        logger.warning(f"无法读取图片尺寸: {image_path} — {e}")
        return False

    return height > LONG_IMAGE_THRESHOLD and ocr_text_len < LOW_OCR_TEXT_THRESHOLD


class LongImageSlicer:
    """长图切片器"""

    def __init__(
        self,
        data_dir: str = "data",
        slice_height: int = SLICE_HEIGHT,
        overlap: int = SLICE_OVERLAP,
        max_width: int = MAX_WIDTH,
        max_slices: int = MAX_SLICES,
    ):
        self.data_dir = Path(data_dir)
        self.slice_dir = self.data_dir / "slices"
        self.slice_dir.mkdir(parents=True, exist_ok=True)
        self.slice_height = slice_height
        self.overlap = overlap
        self.max_width = max_width
        self.max_slices = max_slices

    def slice_image(self, image_path: str | Path, image_index: int = 0) -> list[ImageSlice]:
        """对单张图片进行切片

        Args:
            image_path: 图片本地路径
            image_index: 原图在文章中的索引

        Returns:
            切片列表(按 y_start 升序)
        """
        image_path = Path(image_path)
        with Image.open(image_path) as img:
            width, height = img.size

            # 宽度缩放
            scale = 1.0
            if width > self.max_width:
                scale = self.max_width / width
                new_width = self.max_width
                new_height = int(height * scale)
                img = img.resize((new_width, new_height), Image.LANCZOS)
                logger.info(f"图片缩放: {width}x{height} → {new_width}x{new_height}")
                height = new_height

            # 计算切片数
            step = self.slice_height - self.overlap
            num_slices = max(1, (height + step - 1) // step)  # ceil
            if num_slices > self.max_slices:
                num_slices = self.max_slices
                logger.warning(f"切片数 {num_slices} 超过上限 {self.max_slices},截断")

            slices: list[ImageSlice] = []
            stem = image_path.stem

            for i in range(num_slices):
                y_start = i * step
                y_end = min(y_start + self.slice_height, height)

                # 确保最后一个切片覆盖到底部
                if i == num_slices - 1:
                    y_end = height
                    y_start = max(0, y_end - self.slice_height)

                # 裁剪
                crop = img.crop((0, y_start, img.width, y_end))

                # 保存
                slice_filename = f"{stem}_slice_{image_index}_{i}.jpg"
                slice_path = self.slice_dir / slice_filename
                crop.save(slice_path, "JPEG", quality=90)

                # 是否为底部切片
                is_bottom = (y_end >= height - 100)  # 底部 100px 内

                meta = SliceMeta(
                    image_index=image_index,
                    slice_index=i,
                    y_start=y_start,
                    y_end=y_end,
                    is_bottom=is_bottom,
                )
                slices.append(ImageSlice(
                    pil_image=crop,
                    local_path=str(slice_path),
                    meta=meta,
                ))

            logger.info(f"切片完成: {image_path.name} → {len(slices)} 片")
            return slices
```

"""预算管理器(SRS §8.1 第3层门控 — 预算硬上限)

budget.daily_vlm_budget_cny 耗尽即停止 VLM,剩余转 need_review。
"""
from __future__ import annotations

from loguru import logger


class BudgetManager:
    """VLM 预算管理器"""

    def __init__(
        self,
        daily_budget_cny: float = 5.0,
        max_slices_per_article: int = 8,
    ):
        self.daily_budget = daily_budget_cny
        self.max_slices_per_article = max_slices_per_article
        self._spent: float = 0.0
        self._vlm_calls: int = 0
        self._total_slices: int = 0

    @property
    def remaining(self) -> float:
        """剩余预算"""
        return max(0.0, self.daily_budget - self._spent)

    @property
    def total_vlm_calls(self) -> int:
        return self._vlm_calls

    @property
    def total_slices(self) -> int:
        return self._total_slices

    def is_exhausted(self) -> bool:
        """预算是否已耗尽"""
        return self.remaining <= 0

    def slice_limit_reached(self) -> bool:
        """切片数是否已达上限"""
        return self._total_slices >= self.max_slices_per_article

    def can_afford(self, cost: float) -> bool:
        """判断剩余预算是否足够支付"""
        return self.remaining >= cost

    def consume(self, cost: float, slices: int = 1) -> None:
        """消费预算

        Args:
            cost: 本次 VLM 调用花费(元)
            slices: 本次处理的切片数
        """
        self._spent += cost
        self._vlm_calls += 1
        self._total_slices += slices
        logger.info(
            f"VLM 预算消费: {cost:.4f} 元 ({slices} 切片), "
            f"累计 {self._spent:.4f}/{self.daily_budget:.2f} 元"
        )

    def reset(self) -> None:
        """重置预算(新的一天)"""
        self._spent = 0.0
        self._vlm_calls = 0
        self._total_slices = 0
        logger.info("VLM 预算已重置")

    def summary(self) -> dict:
        """获取预算摘要"""
        return {
            "daily_budget": self.daily_budget,
            "spent": round(self._spent, 4),
            "remaining": round(self.remaining, 4),
            "vlm_calls": self._vlm_calls,
            "total_slices": self._total_slices,
        }
```

```python
"""VLM 切片结果合并去重(PRD §8.3.2 / SRS §4.4)

合并规则:
- 岗位唯一键 = normalize(company) + normalize(job_name) + normalize(location)
- 相邻切片重复岗位 → 保留字段更完整者
- 邮箱冲突 → 保留正则合法且置信度高者,标记 conflict
"""
from __future__ import annotations

import re

from wehire_monitor.domain.models import Job

_EMAIL_RE = re.compile(r"^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$")


def _normalize(text: str | None) -> str:
    """归一化文本(去空格/转小写)"""
    if not text:
        return ""
    return text.strip().lower().replace(" ", "")


def _job_key(job: Job) -> str:
    """岗位唯一键"""
    return f"{_normalize(job.company_name)}|{_normalize(job.job_name)}|{_normalize(job.location)}"


def _job_completeness(job: Job) -> int:
    """字段完整度评分(越高越完整)"""
    score = 0
    if job.company_name:
        score += 1
    if job.job_name:
        score += 1
    if job.location:
        score += 1
    if job.email:
        score += 2
    if job.apply_channel:
        score += 1
    if job.deadline and job.deadline.date:
        score += 1
    return score


def merge_slice_jobs(slice_jobs: list[list[Job]]) -> list[Job]:
    """合并多个切片的岗位列表,去重

    Args:
        slice_jobs: 每个切片提取的岗位列表(按切片顺序)

    Returns:
        合并去重后的岗位列表(保持首次出现顺序)
    """
    merged: list[Job] = []
    seen: dict[str, int] = {}  # job_key → 在 merged 中的索引

    for jobs in slice_jobs:
        for job in jobs:
            key = _job_key(job)
            if not key or key == "||":
                # 空键直接保留(可能是无效数据)
                merged.append(job)
                continue

            if key not in seen:
                # 首次出现
                seen[key] = len(merged)
                merged.append(job)
            else:
                # 重复出现,合并
                idx = seen[key]
                existing = merged[idx]
                merged[idx] = _merge_two(existing, job)

    return merged


def _merge_two(a: Job, b: Job) -> Job:
    """合并两个相同岗位,保留字段更完整者"""
    # 选择基础: 完整度 > 置信度
    if _job_completeness(b) > _job_completeness(a):
        base, other = b, a
    else:
        base, other = a, b

    # 检查邮箱冲突
    if base.email and other.email and base.email != other.email:
        # 邮箱冲突: 保留正则合法且置信度高者
        if _EMAIL_RE.match(other.email) and not _EMAIL_RE.match(base.email):
            base.email = other.email
            base.email_chars = other.email_chars
        # 标记冲突
        if "_warnings" not in base.source_evidence:
            base.source_evidence["_warnings"] = []
        if "email_conflict" not in base.source_evidence["_warnings"]:
            base.source_evidence["_warnings"].append("email_conflict")
    elif not base.email and other.email:
        # base 缺邮箱,从 other 补充
        base.email = other.email
        base.email_chars = other.email_chars

    # 补充缺失字段
    if not base.apply_channel and other.apply_channel:
        base.apply_channel = other.apply_channel
    if not base.deadline.date and other.deadline.date:
        base.deadline = other.deadline

    # 置信度取较高者
    base.confidence = max(base.confidence, other.confidence)

    return base
```

```python
"""Extractor — AI 混合提取(SRS §4.4)

三路切换逻辑(v0.3):
  正文>=500字 且含岗位/投递信息 → 文本 LLM
  否则有图片 → OCR → 质量评分
    score>=0.72 → OCR文本 + 原文 → 文本 LLM
    0.45<=score<0.72 → VLM 复核(低成本,关键切片)
    score<0.45 或长图复杂 → 长图切片 → VLM
  预算耗尽 → need_review
"""
from __future__ import annotations

from loguru import logger

from wehire_monitor.domain.models import (
    OCRResult,
    ParsedArticle,
    PrefilterResult,
    ExtractionResult,
)
from wehire_monitor.modules.extractor.ocr_quality import calculate_ocr_quality
from wehire_monitor.modules.extractor.postprocess import postprocess_jobs
from wehire_monitor.modules.extractor.slicer import should_slice_image, LongImageSlicer
from wehire_monitor.modules.extractor.vlm_merge import merge_slice_jobs

# 触发文本 LLM 的关键词
_JOB_TRIGGER_WORDS = {"招聘", "岗位", "职位", "投递", "简历", "报名", "应聘", "录用"}
_TEXT_LLM_THRESHOLD = 500  # 正文字数阈值

# OCR 质量评分阈值
_OCR_QUALITY_HIGH = 0.72   # >= 走 OCR文本+原文→文本 LLM
_OCR_QUALITY_MID = 0.45    # >= 走 VLM 复核; < 走 VLM 切片


class Extractor:
    """AI 混合提取器(三路切换)"""

    def __init__(
        self,
        llm_provider,
        ocr_provider=None,
        vlm_provider=None,
        slicer: LongImageSlicer | None = None,
        budget_manager=None,
    ):
        self.llm = llm_provider
        self.ocr = ocr_provider
        self.vlm = vlm_provider
        self.slicer = slicer
        self.budget = budget_manager

    def extract(
        self, article: ParsedArticle, prefilter: PrefilterResult
    ) -> ExtractionResult:
        """执行提取,返回 ExtractionResult"""
        text = article.plain_text
        has_images = bool(article.images)
        has_job_keywords = any(kw in text for kw in _JOB_TRIGGER_WORDS)
        text_long_enough = len(text) >= _TEXT_LLM_THRESHOLD

        # 路径1: 正文足够长且含岗位/投递信息 → 直接文本 LLM
        if text_long_enough and has_job_keywords:
            logger.info(f"文章 {article.article_id[:8]}: 文本 LLM 路径(正文{len(text)}字)")
            return self._extract_with_llm(article)

        # 路径2: 有图片 → OCR → 质量评分 → 三路分流
        if has_images and self.ocr is not None:
            logger.info(f"文章 {article.article_id[:8]}: OCR 路径({len(article.images)}张图)")
            return self._extract_with_ocr(article)

        # 路径3: 正文短且无图 → 仍尝试文本 LLM
        logger.info(f"文章 {article.article_id[:8]}: 文本 LLM 路径(兜底,{len(text)}字)")
        return self._extract_with_llm(article)

    def _extract_with_llm(
        self,
        article: ParsedArticle,
        ocr_text: str = "",
        ocr_calls: int = 0,
    ) -> ExtractionResult:
        """调用文本 LLM 提取"""
        combined_text = article.plain_text
        if ocr_text:
            combined_text = f"{article.plain_text}\n\n[图片OCR文本]\n{ocr_text}"

        publish_time = article.publish_time or ""

        response = self.llm.extract_jobs(
            text=combined_text,
            title=article.title,
            publish_time=publish_time,
        )

        if not response.success:
            logger.error(f"LLM 提取失败: {response.error}")
            return ExtractionResult(
                article_type="unknown",
                jobs=[],
                warnings=[f"LLM error: {response.error}"],
                llm_calls=1,
                ocr_calls=ocr_calls,
            )

        fallback_date = publish_time[:10] if publish_time else ""
        jobs = postprocess_jobs(response.jobs, fallback_date, article_text=combined_text)

        return ExtractionResult(
            article_type=response.article_type,
            jobs=jobs,
            warnings=list(response.warnings),
            llm_calls=1,
            ocr_calls=ocr_calls,
        )

    def _extract_with_ocr(self, article: ParsedArticle) -> ExtractionResult:
        """OCR → 质量评分 → 三路分流(文本LLM / VLM复核 / VLM切片)"""
        all_lines = []
        ocr_texts: list[str] = []
        ocr_calls = 0
        y_offset = 0

        for img in article.images:
            if not img.local_path:
                continue
            result = self.ocr.ocr(img.local_path)
            ocr_calls += 1
            if result.lines:
                from wehire_monitor.domain.models import OCRLine
                for line in result.lines:
                    adjusted_box = [
                        line.box[0],
                        line.box[1] + y_offset,
                        line.box[2],
                        line.box[3],
                    ]
                    all_lines.append(OCRLine(
                        text=line.text,
                        confidence=line.confidence,
                        box=adjusted_box,
                    ))
                max_y = max(
                    (line.box[1] + line.box[3] for line in result.lines),
                    default=y_offset,
                )
                y_offset = max_y + 10
            if result.full_text:
                ocr_texts.append(result.full_text)

        combined_ocr = "\n".join(ocr_texts)

        if not combined_ocr:
            logger.warning("OCR 无文本,尝试 VLM 或回退文本 LLM")
            return self._try_vlm_or_fallback(article, ocr_calls, combined_ocr)

        ocr_result = OCRResult(lines=all_lines, full_text=combined_ocr)
        quality = calculate_ocr_quality(ocr_result)
        logger.info(f"OCR 质量评分: {quality:.2f}")

        if quality >= _OCR_QUALITY_HIGH:
            # 高质量: OCR文本 + 原文 → 文本 LLM
            return self._extract_with_llm(
                article, ocr_text=combined_ocr, ocr_calls=ocr_calls
            )
        elif quality >= _OCR_QUALITY_MID:
            # 中质量: VLM 复核
            logger.info(f"OCR 质量中等({quality:.2f}),走 VLM 复核")
            return self._extract_with_vlm(article, ocr_calls, combined_ocr, force_slice=False)
        else:
            # 低质量: 检查是否长图 → VLM 切片
            logger.warning(f"OCR 质量过低({quality:.2f}),尝试 VLM 切片")
            return self._extract_with_vlm(article, ocr_calls, combined_ocr, force_slice=True)

    def _try_vlm_or_fallback(
        self, article: ParsedArticle, ocr_calls: int, ocr_text: str
    ) -> ExtractionResult:
        """OCR 无文本时尝试 VLM 或回退文本 LLM"""
        if self.vlm is not None and self.budget and not self.budget.is_exhausted():
            return self._extract_with_vlm(article, ocr_calls, ocr_text, force_slice=True)
        return self._extract_with_llm(article, ocr_calls=ocr_calls)

    def _extract_with_vlm(
        self,
        article: ParsedArticle,
        ocr_calls: int,
        ocr_text: str,
        force_slice: bool = False,
    ) -> ExtractionResult:
        """VLM 提取(切片或复核)

        Args:
            force_slice: True=强制切片(低质量); False=复核(中质量)
        """
        if self.vlm is None:
            # 无 VLM provider,回退文本 LLM 或 need_review
            if force_slice:
                return ExtractionResult(
                    article_type="unknown", jobs=[],
                    warnings=["need_review: OCR quality low and no VLM configured"],
                    ocr_calls=ocr_calls,
                )
            return self._extract_with_llm(article, ocr_text=ocr_text, ocr_calls=ocr_calls)

        # 预算检查
        if self.budget and self.budget.is_exhausted():
            logger.warning("VLM 预算耗尽,标记 need_review")
            return ExtractionResult(
                article_type="unknown", jobs=[],
                warnings=["need_review: VLM budget exhausted"],
                ocr_calls=ocr_calls,
            )

        # 切片处理
        all_slices = []
        if force_slice and self.slicer is not None:
            for img in article.images:
                if not img.local_path:
                    continue
                if should_slice_image(img.local_path, len(ocr_text)):
                    slices = self.slicer.slice_image(img.local_path, image_index=img.index)
                    all_slices.extend(slices)
                else:
                    # 不需要切片的图片,直接作为单张切片
                    from wehire_monitor.domain.models import ImageSlice, SliceMeta
                    all_slices.append(ImageSlice(
                        pil_image=None,
                        local_path=img.local_path,
                        meta=SliceMeta(
                            image_index=img.index, slice_index=0,
                            y_start=0, y_end=img.height,
                        ),
                    ))
        else:
            # 复核模式: 取关键图片(第一张或最大的)
            for img in article.images:
                if not img.local_path:
                    continue
                from wehire_monitor.domain.models import ImageSlice, SliceMeta
                all_slices.append(ImageSlice(
                    pil_image=None,
                    local_path=img.local_path,
                    meta=SliceMeta(
                        image_index=img.index, slice_index=0,
                        y_start=0, y_end=img.height,
                    ),
                ))

        if not all_slices:
            logger.warning("无可用切片,回退文本 LLM")
            return self._extract_with_llm(article, ocr_text=ocr_text, ocr_calls=ocr_calls)

        # 逐切片调用 VLM
        all_jobs = []
        all_warnings = list(ocr_text and [f"ocr_quality: {0:.2f}"] or [])
        total_cost = 0.0
        vlm_calls = 0

        for sl in all_slices:
            if self.budget and not self.budget.can_afford(0.03):
                all_warnings.append("need_review: VLM budget insufficient for remaining slices")
                break
            if self.budget and self.budget.slice_limit_reached():
                all_warnings.append("slice_limit_reached: max_slices_per_article")
                break

            resp = self.vlm.extract_jobs_from_slices(
                slices=[sl],
                title=article.title,
                publish_time=article.publish_time or "",
            )
            vlm_calls += 1

            if resp.success:
                all_jobs.append(resp.jobs)
                total_cost += resp.cost_estimate
                if self.budget:
                    self.budget.consume(cost=resp.cost_estimate, slices=1)
            else:
                all_warnings.append(f"vlm_slice_{sl.meta.slice_index}: {resp.error}")

        # 合并切片结果
        merged_jobs = merge_slice_jobs(all_jobs)

        # 后处理
